Python: Class

Overview


class MyClass:
def __init__(self, name, age):
  """Constructor: Initializes the object with name and age"""
  self.name = name  # Instance variable
  self.age = age

def greet(self):
  """A regular method"""
  return f"Hello, my name is {self.name} and I am {self.age} years old."

@staticmethod
def static_method(a, b):
  """
  A static method does not use self and does not have access
  to the attributes of a class or any of its instances.
  Example usage: Math util classes with useful functions.
  """
  return f"This is a static method. {a + b}"

@classmethod
def class_method(cls, shortenedName="Not sure"):
  """
  A class method (works with the class itself).
  Acts as an alternate constructor for the class.
  """
  print(f"This method belongs to the {cls.__name__} class.")

  age = 25
  name = shortenedName
  if name == "A":
    name = "Alice"
  return MyClass(name, age)
  
# Creating an object (instance of the class)
person = MyClass("Alice", 25)
person2 = MyClass.class_method("A")
# Calling methods
print(person.greet())
print(person2.greet())
print(MyClass.static_method(20, 10))
