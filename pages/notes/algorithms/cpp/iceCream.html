void whatFlavors(vector cost, int money)
{
    int iFirst = 0, iSecond = 0;
    unordered_map mp;  // stores {cost value -> index}

    for (int i = 0; i < cost.size(); ++i)
    {
        iSecond = i;

        // Check if the complement (money - current cost) exists
        if (mp.count(money - cost[i]))
        {
            iFirst = mp[money - cost[i]];
            cout << iFirst + 1 << " " << iSecond + 1 << endl;  // 1-based index
            return;
        }

        // Store the current cost and its index
        mp[cost[i]] = i;
    }
}

Time complexity analysis

The unordered map ensures O(1) average time complexity for lookups.
