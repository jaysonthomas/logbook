int maximumToys(vector prices, int k)
{
  int n, temp, toys;
  int budget, smallest_index;
  n = prices.size();
  toys = 0;
  budget = k;
  smallest_index = n - 1;
  while (budget > 0)
  {
    for (int i = 0; i < smallest_index; i++)
    {
      if (prices[i] < prices[i + 1])
      {
        temp = prices[i + 1];
        prices[i + 1] = prices[i];
        prices[i] = temp;
      }
    }

    if (budget >= prices[smallest_index])
    {
      budget -= prices[smallest_index];
      toys++;
    }
    else
    {
      budget = -1;
    }
    smallest_index--;
  }
  return toys;
}
