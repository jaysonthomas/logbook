using namespace std;

string twoStrings(string s1, string s2)
{
    unordered_set chars_in_s1(s1.begin(), s1.end());

    for (char c : s2)
    {
        if (chars_in_s1.count(c))
            return "YES";
    }

    return "NO";
}
All characters of s1 is placed in a hash set. Then, for each character in s2, it checks if that character is in the set.
