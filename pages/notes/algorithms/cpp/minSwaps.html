[Ref]

The unordered array contains numbers from 1 (not 0) to 5, if the size of the array is 5 for example. 

int minimumSwaps(vector arr)
{
  int iRet = 0;
  for (int i = 0; i < arr.size(); ++i)
  {
    if (arr[i] == (i + 1))
    {
      continue;
    }
    for (int j = i + 1; j < arr.size(); ++j)
    {
      if (arr[j] == (i + 1))
      {
        std::swap(arr[i], arr[j]);
        iRet++;
        break;
      }
    }
  }
  return iRet;
}
