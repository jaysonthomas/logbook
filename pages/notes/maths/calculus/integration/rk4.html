$$\begin{align*} \newcommand{\oState}[1]{\begin{bmatrix}\dot\theta_{#1} \\ \ddot\theta_{#1} \end{bmatrix}} \newcommand{\iState}{\begin{bmatrix}\theta \\ \dot\theta \end{bmatrix}} \newcommand{\iPrevOutput}[1]{ \begin{bmatrix} \theta + \dot\theta_{#1} \cdot \frac{h}{2} \\ \dot\theta + \ddot\theta_{#1} \cdot \frac{h}{2} \\ \end{bmatrix} } \newcommand{\iPrevOutputTwo}[1]{ \begin{bmatrix} \theta + \dot\theta_{#1} \cdot h \\ \dot\theta + \ddot\theta_{#1} \cdot h \\ \end{bmatrix} } k_1 = \oState{_1} &= f(t, \iState) \\ k_2 = \oState{_2} &= f(t + \frac{h}{2}, \iPrevOutput{1}) \\ k_3 = \oState{_3} &= f(t + \frac{h}{2}, \iPrevOutput{2}) \\ k_4 = \oState{_4} &= f(t + h, \iPrevOutputTwo{3}) \\\\ \theta &= \theta + (\dot\theta_1 + 2\dot\theta_2 + 2\dot\theta_3 + \dot\theta_4) \cdot \frac{h}{6} \\ \dot\theta &= \dot\theta + (\ddot\theta_1 + 2\ddot\theta_2 + 2\ddot\theta_3 + \ddot\theta_4) \cdot \frac{h}{6} \\ t &= t + h \end{align*}$$
